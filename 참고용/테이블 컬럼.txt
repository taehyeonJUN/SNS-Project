CREATE TABLE TEMPMEMBER(
  MEM_ID_TEMP VARCHAR2(20) NOT NULL PRIMARY KEY,
  MEM_PASS_TEMP VARCHAR2(20)NOT NULL,
  MEM_NAME_TEMP VARCHAR2(20) NOT NULL,
  MEM_TEL_TEMP  VARCHAR2(50) NOT NULL,
  MEM_REGDATE_TEMP  date,
  MEM_AUTHOR_TEMP VARCHAR2(10) DEFAULT 'USER',
  CONSTRAINT TEMPMEMBER_CH_MEM_AUTHOR_TEMP CHECK (MEM_AUTHOR_TEMP IN('ADMIN','USER')
));

CREATE TABLE MEMBER(
  MEM_ID VARCHAR2(20) NOT NULL PRIMARY KEY,
  MEM_PASS VARCHAR2(20)NOT NULL,
  MEM_NAME VARCHAR2(20) NOT NULL,
  MEM_TEL  VARCHAR2(50) NOT NULL,
  MEM_REG_DATE  date,
  MEM_AUTHOR VARCHAR2(10) DEFAULT 'USER',
  MEM_THEME_COLOR VARCHAR2(50),
  MEM_PHOTO_ADDR VARCHAR2(200), 
  MEM_INTRO VARCHAR2(200),
  CONSTRAINT MEMBER_CH_MEM_AUTHOR CHECK (MEM_AUTHOR IN('ADMIN','USER'))
);

CREATE TABLE FOLLOWER(
FOLLOWER_NO INT PRIMARY KEY,
FOLLOWER_ID VARCHAR2(20),
FOLLOWER_COUNT INT,
FOREIGN KEY(FOLLOWER_ID) REFERENCES MEMBER(MEM_ID)ON DELETE CASCADE
);

CREATE TABLE FOLLOW(
FOLLOW_NO INT PRIMARY KEY,
FOLLOW_I VARCHAR2(20),
FOLLOW_Y VARCHAR2(20),
FOREIGN KEY(FOLLOW_I) REFERENCES MEMBER(MEM_ID)ON DELETE CASCADE,
FOREIGN KEY(FOLLOW_Y) REFERENCES MEMBER(MEM_ID)ON DELETE CASCADE
);
